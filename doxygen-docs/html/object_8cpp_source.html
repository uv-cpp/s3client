52 ETag S3Api::PutObject(const std::string &bucket, const std::string &key,
53  const CharArray &buffer, Headers headers,
54  const string &payloadHash) {
55 
56  headers.insert({"content-length", to_string(buffer.size())});
57  const auto &wc = Send({.method = "PUT",
58  .bucket = bucket,
59  .key = key,
60  .headers = headers,
61  .payloadHash = payloadHash,
62  .uploadData = buffer.data(),
63  .uploadDataSize = buffer.size()});
64  const string etag = HTTPHeader(wc.GetHeaderText(), "ETag");
65  if (etag.empty()) {
66  throw runtime_error("Missing ETag");
67  }
68  return etag;
69 }

92 ETag S3Api::PutFileObject(const std::string &fileName,
93  const std::string &bucket, const std::string &key,
94  size_t offset, size_t size, Headers headers,
95  const std::string &payloadHash) {
96  const size_t fsize = size ? size : filesystem::file_size(fileName);
97  headers.insert({"content-length", to_string(fsize)});
98  Config({.method = "PUT",
99  .bucket = bucket,
100  .key = key,
101  .headers = headers,
102  .payloadHash = payloadHash});
103  if (!webClient_.UploadFile(fileName, 0, fsize)) {
104  throw runtime_error("Error uploading file - " + webClient_.ErrorMsg());
105  }
106  HandleError(webClient_);
107  return HTTPHeader(webClient_.GetHeaderText(), "ETag");
108 }

124 //------------------------------------------------------------------------------
125 void S3Api::GetObject(const std::string &bucket, const std::string &key,
126  CharArray &buffer, size_t offset, size_t begin,
127  size_t end, Headers headers) {
128  if (end > 0) {
129  headers.insert(
130  {"Range", "bytes=" + to_string(begin) + "-" + to_string(end)});
131  }
132  const auto &wc =
133  Send({.method = "GET", .bucket = bucket, .key = key, .headers = headers});
134  const auto &bytes = wc.GetResponseBody();
135  if (buffer.begin() + offset + bytes.size() >= buffer.end()) {
136  throw range_error("Out buffer too small");
137  }
138  copy(bytes.begin(), bytes.end(), buffer.begin() + offset);
139 }

156 //------------------------------------------------------------------------------
157 void S3Api::GetFileObject(const std::string &fileName,
158  const std::string &bucket, const std::string &key,
159  size_t offset, size_t begin, size_t end,
160  Headers headers) {
161 
162  FILE *out = !filesystem::exists(fileName) ? fopen(fileName.c_str(), "wb")
163  : fopen(fileName.c_str(), "r+b");
164  if (!out) {
165  throw runtime_error("Cannot open file " + fileName + " for writing");
166  }
167  fseek(out, offset, SEEK_SET);
168 
169  if (end > 0) {
170  headers.insert(
171  {"range", "bytes=" + to_string(begin) + "-" + to_string(end)});
172  }
173  Config({.method = "GET", .bucket = bucket, .key = key, .headers = headers});
174  webClient_.SetWriteFunction(nullptr, out);
175  webClient_.Send();
176  fclose(out);
177  HandleError(webClient_);
178 }

206  const ListObjectV2Config &config,
207  const Headers &headers) {

210  params["delimiter"] = config.delimiter;

212  params["fetch-owner"] = config.fetchOwner;
213  // params["max-keys"] = to_string(config.maxKeys);

215  params["start-after"] = config.startAfter;
216  const auto &wc = Send({.method = "GET",
217  .bucket = bucket,
218  .params = params,
219  .headers = headers});
220  return ParseObjects(webClient_.GetContentText());

48 /// Representaion of an XML tree as {path to text element, text element} tuples
49 /// stored into map object. E.g. \code{.xml} <tag1>
50 /// <tag1_1>
51 /// text 1 1
52 /// </tat1_1>

54 /// text 1 2
55 /// </tag1_2>
56 /// </tag1>

58 /// is represented as
59 /// \code{.cpp}
60 /// unordered_map<string, string> rec {

63 /// };
64 /// \endcode

87 /**
88  *
89  * \brief XMLGenerator. Use methods or overloaded subscript operator to insert
90  * data into XML tree.
91  *
92  * A XML document object and a pointer to the current XML Element are maintained
93  * inside the class instance.
94  * Elements are added under the current element.
95  * The current element pointer can be moved up or down the tree as needed.
96  * The generated XML code is extracted by calling the XMLText method.
97  *
98  * \section Example
99  * Generating the ACL request body.
100  *
101  * Request format:
102  * \code{.xml}
103  * <AccessControlPolicy>
104  * <Owner>
105  * <DisplayName>string</DisplayName>
106  * <ID>string</ID>
107  * </Owner>
108  * <AccessControlList>
109  * <Grant>
110  * <Grantee>
111  * <DisplayName>string</DisplayName>
112  * <EmailAddress>string</EmailAddress>
113  * <ID>string</ID>
114  * <xsi:type>string</xsi:type>
115  * <URI>string</URI>
116  * </Grantee>
117  * <Permission>string</Permission>
118  * </Grant>
119  * </AccessControlList>
120  * </AccessControlPolicy>
121  * \endcode
122  * Code:
123  * \snippet api/xml_parser.cpp GenerateAclXML
124  */
125 class XMLOStream {
126 public:
127  /// Move up or down one level in tree or rewind to root.
128  enum MoveAction { UP, DOWN, REWIND };
129  /// Move up
130  /// \param[in] level number of parents to traverse, default is one
131  void Up(int level = 1) {
132  if (!cur_) {
133  throw std::logic_error("Cannot pop, Null element");
134  }
135  if (!cur_->Parent()) {
136  throw std::logic_error("Cannot pop, Null parent");
137  }
138  for (; level; cur_ = cur_->Parent()->ToElement(), level--)
139  ;
140  }
141  /// Move down after setting text
142  /// Set \c down_ flag: pointer to current node will be moved to child after
143  /// next insertion
144  void Down() { down_ = true; }
145  /// Move pointer to current node to root.
146  void Rewind() {
147  if (!cur_) {
148  return;
149  }
150  for (; cur_->Parent(); cur_ = cur_->Parent()->ToElement())
151  ;
152  }
153  /// Insert text inside current element.
154  /// \param[in] text text to insert
155  /// \return reference to current \c XMLOStream instance, will return
156  /// without inserting in case current pointer to XML element is \c NULL
157  XMLOStream &InsertText(const std::string &text) {
158  if (!cur_)
159  return *this;
160  cur_->SetText(text.c_str());
161  return *this;
162  }
163  /// Insert XML element under current XML element.
164  /// \param[in] s tag name
165  /// \return reference to current \c XMLOStream instance
166  XMLOStream &Insert(const std::string &s) {
167  std::vector<std::string> elems;
168  std::istringstream is(s);
169  for (std::string buf; std::getline(is, buf, '/');) {
170  if (!buf.empty()) {
171  elems.push_back(buf);
172  }
173  }
174  if (elems.empty())
175  return *this;
176  auto pe = cur_;
177  if (!cur_) {
178  cur_ = doc_.NewElement(elems.front().c_str());
179  doc_.InsertFirstChild(cur_);
180  pe = cur_;
181  } else {
182  pe = cur_->InsertNewChildElement(s.c_str());
183  }
184  for (auto e = ++begin(elems); e != end(elems); ++e) {
185  pe = pe->InsertNewChildElement(e->c_str());
186  }
187  if (down_) {
188  cur_ = pe;
189  down_ = false;
190  }
191  return *this;
192  }
193  /// Move pointer to XML element.
194  /// \see MoveAction
195  /// In the case of \c REWIND or \c UP the action is executed immediately
196  /// in case of \c DOWN it is executed after next insertion operation
197  XMLOStream &Move(MoveAction a, int level = 1) {
198  switch (a) {
199  case UP:
200  Up(level);
201  break;
202  case DOWN:
203  Down();
204  break;
205  case REWIND:
206  Rewind();
207  break;
208  default:
209  break;
210  }
211  return *this;
212  }
213 
214  /// Invokes insert method forcing \c DOWN action after insert.
215  /// In case only path separator (\c '/') elements are provided the \c UP
216  /// action is invoked as many times as there are \c '/' caracters in the
217  /// string.
218  XMLOStream &operator[](const std::string &s) {
219  if (all_of(begin(s), end(s), [](auto c) { return c == '/'; })) {
220  Up(s.size());
221  return *this;
222  }
223  down_ = true;
224  return Insert(s);
225  }
226  /// Invoke \c Insert(MoveAction) method
227  XMLOStream &operator[](MoveAction a) { return Move(a); }
228  /// Constructor: requires a reference to a pre-existing XML document.
229  /// \param[in] d wrapped XML document
230  XMLOStream(tinyxml2::XMLDocument &d) : doc_(d) {}
231  /// Add text element to current XML element and move up one level.
232  /// \param[in] s text to insert
233  /// \return reference to XMLOStream instance
234  XMLOStream &operator=(const std::string &s) {
235  InsertText(s);
236  Up();
237  return *this;
238  }
239  /// \return current content of XML document as XML text
240  std::string XMLText() const { return XMLToText(doc_, true, 0, 0); }
241  /// \return reference to current document
242  const tinyxml2::XMLDocument &GetDocument() const { return doc_; }
243 
244 private:
245  tinyxml2::XMLDocument &doc_;
246  tinyxml2::XMLElement *cur_ = nullptr;
247  bool down_ = false;
248 };

250 //-----------------------------------------------------------------------------
251 /** \brief XML parser, returns array of text elements with the same parent path
252  * or an array of parsed XML sub trees.
253  *
254  * An XML subtree is represented as record in the form
255  * \code
256  * map<path to text element, text element>
257  * \endcode
258  * under the specifed path.
259  *
260  * This class stores data of different types inside a \c variant object and
261  * extracts the value when the class instance is converted to a type during
262  * assignment operations.
263  *
264  * Since the returned types all have empty values, if the path or text is not
265  * found an empty \c std::string, \c std::vector or \c std::unordered_map
266  * is returned.
267  *
268  * \see XMLResult
269  *
270  */
271 class XMLIStream {
272 public:
273  /// Extract array of text elements under specified path or
274  /// list of records in the form `map<path to text element, text element>`.
275  ///
276  /// \param[in] p path to text elements if path starts with \c '/' or
277  /// to sbtrees under path if it does not start with \c '/'
278  ///
279  /// \return reference to current instance
280  /// \see RecordList
281  const XMLIStream &operator[](const std::string &p) const {
282  // return all text elements under path
283  if (p.front() == '/') {
284  auto r = dd_.find(p);
285  if (r == end(dd_)) {
286  v_ = false;
287  return *this;
288  }
289  if (r->second.size() == 1) {
290  v_ = r->second.front();
291  } else {
292  v_ = r->second;
293  }
294  // return all records under path
295  } else {
296  auto r = RecordList("/" + p, dd_);
297  if (r.empty()) {
298  v_ = false;
299  } else {
300  v_ = r;
301  }
302  }
303  return *this;
304  }
305  /// \return current element text or empty string
306  operator std::string() const {
307  if (auto p = std::get_if<std::string>(&v_)) {
308  return *p;
309  } else {
310  return "";
311  }
312  }
313  /// \return array of text elements or empty array
314  operator std::vector<std::string>() const {
315  if (auto p = std::get_if<std::vector<std::string>>(&v_)) {
316  return *p;
317  } else {
318  return {};
319  }
320  }
321  /// \return array of parsed records or empty array
322  operator std::vector<XMLRecord>() const {
323  if (auto p = std::get_if<std::vector<XMLRecord>>(&v_)) {
324  return *p;
325  } else {
326  return {};
327  }
328  }
329  /// Constructor
330  /// \param[in] xml xml text
331  XMLIStream(const std::string &xml) : dd_(DOMToDict(xml)) {}
332 
333 private:
334  mutable XMLResult v_;
335  std::unordered_map<std::string, std::vector<std::string>> dd_;
336 };

340 /// \param[in] r XMLRecord instance \see XMLRecord

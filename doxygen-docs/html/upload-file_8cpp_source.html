54 //-----------------------------------------------------------------------------
55 ETag DoUploadPart(S3Api &s3, const string &file, size_t offset, size_t size,
56  const string &bucket, const string &key, UploadId uid,
57  int part, int maxRetries) {
58 
59  try {
60  return s3.UploadFilePart(file, offset, size, bucket, key, uid, part);
61  } catch (const exception &e) {
62  if (retriesG++ < maxRetries) {
63  return DoUploadPart(s3, file, offset, size, bucket, key, uid, part,
64  maxRetries);
65  } else {
66  throw e;
67  }
68  }
69 }

71 //-----------------------------------------------------------------------------
72 ETag DoUploadPart(S3Api &s3, const char *data, size_t offset, size_t size,
73  const string &bucket, const string &key, UploadId uid,
74  int part, int maxRetries) {
75 
76  try {
77  return s3.UploadPart(bucket, key, uid, part, data + offset, size,
78  maxRetries);
79  } catch (const exception &e) {
80  if (retriesG++ < maxRetries) {
81  return DoUploadPart(s3, data, offset, size, bucket, key, uid, part,
82  maxRetries);
83  } else {
84  throw e;
85  }
86  }
87 }
88 //-----------------------------------------------------------------------------
89 vector<string> UploadParts(const S3DataTransferConfig &cfg,
90  const string &uploadId, size_t chunkSize,
91  int firstPart, int lastPart, size_t fileSize,
92  int jobId) {
93 
94  S3Api s3(cfg.accessKey, cfg.secretKey,
95  cfg.endpoints[RandomIndex(0, cfg.endpoints.size() - 1)]);
96 
97  vector<string> etags;
98  size_t offset = jobId * chunkSize;
99  chunkSize = min(chunkSize, fileSize - offset);
100  const int numParts = lastPart - firstPart;
101  const size_t partSize = (chunkSize + numParts - 1) / numParts;
102  for (int i = 0; i != numParts; ++i) {
103  const size_t size = min(partSize, chunkSize - i * partSize);
104  etags.push_back(DoUploadPart(s3, cfg.data ? cfg.data : cfg.file, offset,
105  size, cfg.bucket, cfg.key, uploadId,
106  firstPart + i, cfg.maxRetries));
107  offset += size;
108  }
109  return etags;
110 }

112 //-----------------------------------------------------------------------------
113 string UploadFile(const S3DataTransferConfig &cfg, const MetaDataMap &metaData,
114  bool sync) {
115  retriesG = 0;
116  FILE *inputFile = fopen(cfg.file.c_str(), "rb");
117  if (!inputFile) {
118  throw runtime_error(string("cannot open file ") + cfg.file);
119  }
120  fclose(inputFile);
121  // retrieve file size
122  const size_t fileSize = sss::FileSize(cfg.file);
123  if (cfg.endpoints.empty())
124  throw std::logic_error("Missing endpoint information");
125  const string endpoint =
126  cfg.endpoints[RandomIndex(0, cfg.endpoints.size() - 1)];
127  S3Api s3(cfg.accessKey, cfg.secretKey, endpoint);
128  // begin uplaod request -> get upload id
129  const auto uploadId =
130  s3.CreateMultipartUpload(cfg.bucket, cfg.key, 0, metaData);
131 
132  // per-job part size
133  const size_t perJobSize = (fileSize + cfg.jobs - 1) / cfg.jobs;
134  // send parts in parallel and store ETags
135  vector<future<vector<string>>> etags(cfg.jobs);
136  for (int i = 0; i != cfg.jobs; ++i) {
137  etags[i] = async(sync ? launch::deferred : launch::async, UploadParts, cfg,
138  uploadId, perJobSize, i * cfg.partsPerJob,
139  i * cfg.partsPerJob + cfg.partsPerJob, fileSize, i);
140  }
141  vector<ETag> vetags;
142  for (auto &f : etags) {
143  const auto &w = f.get();
144  for (const auto &i : w) {
145  vetags.push_back(i);
146  }
147  }
148  return s3.CompleteMultipartUpload(uploadId, cfg.bucket, cfg.key, vetags);
149 }

151 //-----------------------------------------------------------------------------
152 string UploadData(const S3DataTransferConfig &cfg, const MetaDataMap &metaData,
153  bool sync) {
154  retriesG = 0;
155  if (!cfg.data) {
156  throw logic_error("NULL data buffer");
157  }
158  if (cfg.endpoints.empty())
159  throw std::logic_error("Missing endpoint information");
160  const string endpoint =
161  cfg.endpoints[RandomIndex(0, cfg.endpoints.size() - 1)];
162  S3Api s3(cfg.accessKey, cfg.secretKey, endpoint);
163  // begin uplaod request -> get upload id
164  const auto uploadId =
165  s3.CreateMultipartUpload(cfg.bucket, cfg.key, 0, metaData);
166 
167  // per-job part size
168  const size_t perJobSize = (cfg.size + cfg.jobs - 1) / cfg.jobs;
169  // send parts in parallel and store ETags
170  vector<future<vector<string>>> etags(cfg.jobs);
171  for (int i = 0; i != cfg.jobs; ++i) {
172  etags[i] = async(sync ? launch::deferred : launch::async, UploadParts, cfg,
173  uploadId, perJobSize, i * cfg.partsPerJob,
174  i * cfg.partsPerJob + cfg.partsPerJob, cfg.size, i);
175  }
176  vector<ETag> vetags;
177  for (auto &f : etags) {
178  const auto &w = f.get();
179  for (const auto &i : w) {
180  vetags.push_back(i);
181  }
182  }
183  return s3.CompleteMultipartUpload(uploadId, cfg.bucket, cfg.key, vetags);
184 }

186 //-----------------------------------------------------------------------------
187 string Upload(const S3DataTransferConfig &cfg, const MetaDataMap &metaData,
188  bool sync) {
189 
190  if (cfg.data) {
191  if (!cfg.size) {
192  throw logic_error("Zero size for upload data buffer");
193  }
194  return UploadData(cfg, metaData, sync);
195  } else {
196  if (cfg.file.empty()) {
197  throw logic_error("Empty file name");
198  }
199  return UploadFile(cfg, metaData, sync);
200  }

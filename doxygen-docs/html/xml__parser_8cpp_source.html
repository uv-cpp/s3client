48 std::vector<BucketInfo> ParseBuckets(const std::string &xml) {
49  if (xml.empty())
50  return {};
51  auto d = DOMToDict(xml);
52  if (d.empty()) {
53  return {};
54  }
55  // always use lowercase when case insensitive (default) enabled
56  auto r = ExtractSubPaths("/listallmybucketsresult/buckets", d);
57  if (r.empty()) {
58  return {};
59  }
60  vector<BucketInfo> ret;
61  auto numBuckets = r["/bucket/creationdate"].size();
62  if (numBuckets != r["/bucket/name"].size()) {
63  throw std::logic_error("Malformed response: number of creation dates does "
64  "not match number of buckets");
65  return {};
66  }
67  for (size_t i = 0; i != numBuckets; ++i) {
68  ret.push_back({.name = r["/bucket/name"][i],
69  .creationDate = r["/bucket/creationdate"][i]});
70  }
71  return ret;
72 }

74 std::string Trim(const std::string &text) {
75  auto b = cbegin(text);
76  while (isspace(*b) && (b++ != cend(text)))
77  ;
78  auto e = --cend(text);
79  while (isspace(*e) && (e-- != cbegin(text)))
80  ;
81  return std::string(b, ++e);
82 }

116 // <KeyCount>integer</KeyCount>

118 // <NextContinuationToken>string</NextContinuationToken>
119 // <StartAfter>string</StartAfter>
120 // </ListBucketResult>
121 S3Api::ListObjectV2Result ParseObjects(const std::string &xml) {
122  if (xml.empty())
123  return {};
124  XMLIStream is(xml);
125  S3Api::ListObjectV2Result res;
126  const string truncated = is["/istruncated"];
127  res.truncated = ParseBool(truncated);
128  XMLRecords r = is["listbucketresult/contents"];
129  vector<ObjectInfo> v;
130  for (const auto &i : r) {
131  res.keys.push_back(
132  {.checksumAlgo = Get(i, "/checksumalgo"),
133  .key = Get(i, "/key"),
134  .lastModified = Get(i, "/lastmodified"),
135  .etag = Get(i, "/etag"),
136  .size = Get(i, "/size").empty() ? 0 : stoul(Get(i, "/size")),
137  .storageClass = Get(i, "/storageclass"),
138  .ownerDisplayName = Get(i, "/owner/displayname"),
139  .ownerID = Get(i, "/owner/id")});
140  }
141  return res;
142 }

165 AccessControlPolicy ParseACL(const std::string &xml) {
166  if (xml.empty())
167  return {};
168  XMLIStream is(xml);
169  AccessControlPolicy res;
170  const string prefix = "/accesscontrolpolicy";
171  res.ownerDisplayName = is[prefix + "/owner/displayname"];
172  res.ownerID = is[prefix + "/owner/id"];
173  XMLRecords r = is["accesscontrolpolicy/accesscontrollist/grant"];
174  for (auto &i : r) {
175  const Grant g = {{Get(i, "/grantee/displayname"),
176  Get(i, "/grantee/emailaddress"), Get(i, "/grantee/id"),
177  Get(i, "/grantee/type"), Get(i, "/grantee/uri")},
178  Get(i, "/permission")};
179  res.grants.push_back(g);
180  }
181  return res;
182 }

185 /// [GenerateAclXML]
186 std::string GenerateAclXML(const AccessControlPolicy &acl) {
187  XMLDocument doc;
188  XMLOStream os(doc);
189  os["accesscontrolpolicy/accesscontrollist"];
190  if (!acl.ownerDisplayName.empty() || !acl.ownerID.empty()) {
191  os["owner"]; // <accesscontrolpolicy><accesscontrollist><owner>
192  if (!acl.ownerDisplayName.empty()) {
193  //<accesscontrolpolicy><accesscontrollist><owner><displayname>
194  os["displayname"] = acl.ownerDisplayName;
195  //</displayname> automatically move to upper level after assignment
196  }
197  if (!acl.ownerID.empty()) {
198  //<accesscontrolpolicy><accesscontrollist><owner><ownerid>
199  os["ownerid"] = acl.ownerDisplayName;
200  //</ownerid> automatically move to upper level after assignment
201  }
202  os["/"]; // </owner>
203  }
204  for (const auto &g : acl.grants) {
205  if (g.permission.empty()) {
206  throw logic_error("Missing required field 'permission'");
207  }
208  os["grant"]; // <grant>
209  const Grant::Grantee &i = g.grantee;
210  if (!i.Empty()) {
211  os["grantee"]; // <grant><grantee>
212  if (!i.displayName.empty()) {
213  // <grant><grantee><displayname>
214  os["displayname"] = i.displayName;
215  // </displayname>
216  }
217  if (!i.emailAddress.empty()) {
218  // <grant><grantee><emailaddress>
219  os["emailaddress"] = i.emailAddress;
220  // </emailaddress>
221  }
222  if (!i.id.empty()) {
223  // <grant><grantee><id>
224  os["id"] = i.id;
225  // </id>
226  }
227  if (!i.xsiType.empty()) {
228  // <grant><grantee><type>``
229  os["type"] = i.xsiType;
230  // </type>
231  }
232  if (!i.uri.empty()) {
233  // <grant><grantee><uri>
234  os["uri"] = i.uri;
235  // <uri>
236  }
237  }
238  os["/"] // </grantee>
239  // <grant><permission>
240  ["permission"] = g.permission;
241  // </permission>
242  }
243  return os.XMLText();
244 }

39  * \file parallel_download.cpp
40  * \brief Parallel object download download from S3 service

42 /// [Parallel object download]
43 #include "lyra/lyra.hpp"
44 #include "s3-client.h"
45 #include <filesystem>
46 #include <fstream>
47 #include <iostream>
48 #include <string>
49 using namespace std;
50 using namespace sss;
51 using namespace filesystem;
52 
53 //------------------------------------------------------------------------------
54 int main(int argc, char const *argv[]) {
55  try {
56  S3DataTransferConfig config;
57  bool showHelp = false;
58  string endpoint;
59  string endpointsFile;
60  string credentialsFile;
61  string awsProfile;
62  bool overwrite = false;
63  auto cli =
64  lyra::help(showHelp).description("Download file from S3 bucket") |
65  lyra::opt(config.accessKey,
66  "awsAccessKey")["-a"]["--access_key"]("AWS access key")
67  .optional() |
68  lyra::opt(config.secretKey,
69  "awsSecretKey")["-s"]["--secret_key"]("AWS secret key")
70  .optional() |
71  lyra::opt(endpoint, "endpoint")["-e"]["--endpoint"]("Endpoint URL")
72  .optional() |
73  lyra::opt(endpointsFile, "endpointsFile")["-E"]["--endpoints-file"](
74  "File with list of endpoints")
75  .optional() |
76  lyra::opt(config.bucket, "bucket")["-b"]["--bucket"]("Bucket name")
77  .required() |
78  lyra::opt(config.key, "key")["-k"]["--key"]("Key name").required() |
79  lyra::opt(config.file, "file")["-f"]["--file"]("File name").optional() |
80  lyra::opt(credentialsFile, "credentials file")["-c"]["--credentials"](
81  "Credentials file, AWS cli format")
82  .optional() |
83  lyra::opt(awsProfile, "AWS config profile")["-p"]["--profile"](
84  "Profile in AWS config file")
85  .optional() |
86  lyra::opt(config.maxRetries,
87  "Max retries")["-r"]["--retries"]("Max number of retries")
88  .optional() |
89  lyra::opt(config.jobs,
90  "parallel jobs")["-j"]["--jobs"]("Number of parallel jobs")
91  .optional() |
92  lyra::opt(config.partsPerJob,
93  "chunks per job")["-n"]["--parts_per_job"](
94  "Number of parts per job")
95  .optional() |
96  lyra::opt(overwrite)["-y"]["--overwrite"](
97  "Overwrite exsisting file, default is 'false'")
98  .optional();
99  if (showHelp) {
100  cout << cli;
101  return 0;
102  }
103  // Parse program arguments:
104  auto result = cli.parse({argc, argv});
105  if (!result) {
106  cerr << result.message() << endl;
107  cerr << cli << endl;
108  exit(1);
109  }
110  if (config.file.empty())
111  config.file = config.key;
112  if (exists(config.file) && !overwrite) {
113  cerr << "File exists, use the '-y' command line switch to overwrite"
114  << endl;
115  exit(EXIT_FAILURE);
116  }
117  if (!endpoint.empty()) {
118  config.endpoints.push_back(endpoint);
119  } else {
120  if (endpointsFile.empty()) {
121  cerr << "Specify either an endpoing or a file to read endpoints from"
122  << endl;
123  exit(EXIT_FAILURE);
124  }
125  ifstream is(endpointsFile);
126  if (!is) {
127  cerr << "Cannot read from " << endpointsFile << endl;
128  exit(EXIT_FAILURE);
129  }
130  string line;
131  while (getline(is, line)) {
132  TrimLine(line);
133  if (line.empty() || line[0] == '#')
134  continue;
135  config.endpoints.push_back(line);
136  }
137  }
138  if (config.accessKey.empty() || config.secretKey.empty()) {
139  if (credentialsFile.empty()) {
140  credentialsFile = GetHomeDir() + "/.aws/credentials";
141  }
142  auto c = GetS3Credentials(credentialsFile, awsProfile);
143  config.accessKey = c.accessKey;
144  config.secretKey = c.secretKey;
145  }
146  Download(config);
147  return 0;
148  } catch (const exception &e) {
149  cerr << e.what() << endl;
150  return 1;
151  }
152 }

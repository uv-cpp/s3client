145 /**
146  * \brief S3 Client inteface.
147  *
148  * Implements some of the S3 actions documented here:
149  * https://docs.aws.amazon.com/AmazonS3/latest/API/API_Operations_Amazon_Simple_Storage_Service.html
150  *
151  * Also includes higher level methods to simplify access to the API.
152  *
153  * When sending requests, the fields specified as URL prameters are passed
154  * to methods as C++ structs and sent as part of the request, but in most cases
155  * the optional parameters sent as HTTP header fields are not sent along, it is
156  * therefore required to explicitly pass them in the \c headers parameter.
157  *
158  * E.g. the \c ListObjectV2 request supports HTTP
159  * headers for specifying the payer for the request and owner of the listed
160  * objects. \code{.cpp} s3.ListObjectsV2(bucket, config, {
161  * {"x-amz-request-payer", "RequestPayer"},
162  * {"x-amz-expected-bucket-owner", "ExpectedBucketOwner"}
163  * });
164  * \endcode
165  *
166  * <h2>Error handling</h2>
167  *
168  * Currently exceptions are being used to report errors, work is ongoing to
169  * move to result types similar to Rust's \c Result<ResultT,ErrorT>.
170  *
171  * All methods that send requests throw:
172  *
173  * - \c std::runtime_error in case of errors sending the request
174  * - \c std::logic_error when the returned response code is >= 400
175  *
176  *
177  * \section ex1 Examples
178  * From test cases printing results in \c CSV format.
179  *
180  * \b Bucket
181  * \snippet api/bucket-test.cpp CreateBucket
182  * \snippet api/bucket-test.cpp ListBuckets
183  * \snippet api/bucket-test.cpp DeleteBucket
184  * \b Object
185  * \snippet api/object-test.cpp PutObject
186  * \snippet api/object-test.cpp GetObject
187  * \snippet api/object-test.cpp ListObjectsV2
188  * \snippet api/object-test.cpp DeleteObject
189  */

294  /// \brief Clear data and reset read and write functions
295  void Clear() {
296  webClient_.SetPath("");
297  webClient_.SetHeaders({{}});
298  webClient_.SetReqParameters({{}});
299  webClient_.SetPostData("");
300  webClient_.ClearBuffers();
301  webClient_.ResetRWFunctions();

303  /// \brief Send request.
304  /// \param[in] p send parameters \see SendParams
305  /// \return reference to \c this \c S3Api instance.
306  /// \throws std::runtime_error in case of error sending request
307  /// \throws std::logic_error in case of returned error (status >= 400)
308  const WebClient &Send(const SendParams &p) {
309  /// [WebClient::Send]
310  Config(p);
311  if (ToLower(p.method) == "put") {
312  if (p.uploadData)
313  webClient_.UploadDataFromBuffer(p.uploadData, 0, p.uploadDataSize);
314  else
315  webClient_.Send();
316  } else if (ToLower(p.method) == "post") {
317  if (!p.postData.empty()) {
318  if (p.urlEncodePostParams) {
319  webClient_.SetUrlEncodedPostData(ParseParams(p.postData.data()));
320  } else {
321  webClient_.SetPostData(p.postData);
322  }
323  }
324  webClient_.Send();
325  } else {
326  webClient_.Send();
327  }
328  HandleError(webClient_);
329  return webClient_;
330  /// [WebClient::Send]
331  }

337  /// Send method with configurable send and receive functions.
338  ///
339  /// \param[in] params \see SendParams
340  ///
341  /// \param[in] sendFun pointer to function generating the data to send
342  ///
343  /// \param[in] sendUseData pointer to user data passed to \c sendFun at each
344  /// invocation
345  ///
346  /// \param[in] receiveFun pointer to function receiving data
347  ///
348  /// \param[in] receiveUserData pointer to user data passed to \c receiveFun at
349  /// each invocation
350  void Send(const SendParams &params, WebClient::ReadFunction sendFun,
351  void *sendUserData, WebClient::WriteFunction receiveFun,
352  void *receiveUserData) {
353  webClient_.SetWriteFunction(receiveFun, receiveUserData);
354  webClient_.SetReadFunction(sendFun, sendUserData);
355  Send(params);
356  }

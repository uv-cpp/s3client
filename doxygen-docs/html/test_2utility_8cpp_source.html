42 //----------------------------------------------------------------------------
43 Params ParseCmdLine(int argc, char **argv) {
44 
45  auto printUsage = [](char **argv, ostream &os) {
46  cerr << "Usage: " << argv[0] << " <Access environment variable> "
47  << "<Secret environment variable> <URL environment variable>" << endl;
48  };
49  if (argc != 4) {
50  cerr << "Wrong number of arguments" << endl;
51  printUsage(argv, cerr);
52  exit(EXIT_FAILURE);
53  }
54  const string access = getenv(argv[1]) ? getenv(argv[1]) : "";
55  if (access.empty()) {
56  cerr << "Error: cannot read access env var " << argv[1] << endl;
57  exit(EXIT_FAILURE);
58  }
59  const string secret = getenv(argv[2]) ? getenv(argv[2]) : "";
60  if (secret.empty()) {
61  cerr << "Error: cannot read secret env var " << argv[2] << endl;
62  exit(EXIT_FAILURE);
63  }
64  const string url = getenv(argv[3]) ? getenv(argv[3]) : "";
65  if (secret.empty()) {
66  cerr << "Error: cannot read url env var " << argv[3] << endl;
67  exit(EXIT_FAILURE);
68  }
69  return {.access = access, .secret = secret, .url = url};
70 }

98 string Timestamp() {
99  time_t t;
100  time(&t);
101  struct tm *ts;
102  ts = gmtime(&t);
103 
104  const size_t BUFSIZE = 128;
105  vector<char> buf1(BUFSIZE, '\0');
106  strftime(buf1.data(), BUFSIZE, "%Y%m%dT%H%M%SZ", ts);
107  return string(buf1.data());
108 }

111 TempFile OpenTempFile(const string &mode, const std::string &prefix) {
112  const string path =
113  filesystem::temp_directory_path().string() + "/" + prefix + "XXXXXX";
114  vector<char> buf(begin(path), end(path));
115  buf.push_back('\0');
116  const int fd = mkstemp(buf.data());
117  const string tmpPath(begin(buf), end(buf));
118  return {fdopen(fd, mode.c_str()), tmpPath};
119 }

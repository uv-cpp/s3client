50 inline uint32_t Sigma0(uint32_t x) {
51  return right_rotate(x, 2) ^ right_rotate(x, 13) ^ right_rotate(x, 22);
52 }
53 inline uint32_t Sigma1(uint32_t x) {
54  return right_rotate(x, 6) ^ right_rotate(x, 11) ^ right_rotate(x, 25);
55 }

57 // Choose: if c bit == 0 select bit from y else select bit from x
58 inline uint32_t Ch(uint32_t c, uint32_t x, uint32_t y) {
59  return (c & x) | ((~c) & y);
60 }
61 
62 // Majority: if at least two zeroes return 0 else return 1
63 inline uint32_t Maj(uint32_t x, uint32_t y, uint32_t z) {
64  return (x & y) ^ (x & z) ^ (y & z);
65 }

82 // sha256 algorithm, streaming version: receives and updates hash
83 // data is unsigned byte, all other variables are usigned 32 bit int
84 void sha256_stream(uint32_t hash[8], const uint8_t data[], uint64_t length) {
85  uint32_t a, b, c, d, e, f, g, h;
86  uint32_t w[16];
87  uint64_t blocks = length / 64;
88  while (blocks--) {
89  a = hash[0];
90  b = hash[1];
91  c = hash[2];
92  d = hash[3];
93  e = hash[4];
94  f = hash[5];
95  g = hash[6];
96  h = hash[7];
97 
98  for (int i = 0; i != 16; ++i) {
99  w[i] = lshift(data[0], 24) | lshift(data[1], 16) | lshift(data[2], 8) |
100  lshift(data[3], 0);
101  data += 4;
102 
103  const uint32_t tmp1 = h + Sigma1(e) + Ch(e, f, g) + K[i] + w[i];
104  const uint32_t tmp2 = Sigma0(a) + Maj(a, b, c);
105  h = g;
106  g = f;
107  f = e;
108  e = d + tmp1;
109  d = c;
110  c = b;
111  b = a;
112  a = tmp1 + tmp2;
113  }
114 
115  for (int i = 16; i != 64; ++i) {
116  const uint32_t s0 = w[(i + 1) & 0x0f];
117  const uint32_t sigma0 =
118  right_rotate(s0, 7) ^ right_rotate(s0, 18) ^ (s0 >> 3);
119  const uint32_t s1 = w[(i + 14) & 0x0f];
120  const uint32_t sigma1 =
121  right_rotate(s1, 17) ^ right_rotate(s1, 19) ^ (s1 >> 10);
122  w[i & 0xf] += sigma0 + sigma1 + w[(i + 9) & 0xf];
123  const uint32_t tmp1 = w[i & 0xf] + h + Sigma1(e) + Ch(e, f, g) + K[i];
124  const uint32_t tmp2 = Sigma0(a) + Maj(a, b, c);
125  h = g;
126  g = f;
127  f = e;
128  e = d + tmp1;
129  d = c;
130  c = b;
131  b = a;
132  a = tmp1 + tmp2;
133  }
134 
135  hash[0] += a;
136  hash[1] += b;
137  hash[2] += c;
138  hash[3] += d;
139  hash[4] += e;
140  hash[5] += f;
141  hash[6] += g;
142  hash[7] += h;
143  }
144 }

146 // sha256 on single fixed size buffer
147 void sha256(const uint8_t data[], size_t length, uint32_t hash[8]) {
148  init_hash(hash);
149  const static size_t MAX_BUF_SIZE = 4096;
150  thread_local static uint8_t buf[MAX_BUF_SIZE];
151  int stack_buffer = next_div_by(length + 1 + 8, 64) <= MAX_BUF_SIZE;
152  size_t sz = 0;
153  uint8_t *message =
154  alloc_padded(length, length, &sz, stack_buffer ? buf : NULL);
155  memcpy(message, data, length);
156  sha256_stream(hash, message, sz);
157  if (!stack_buffer)
158  free(message);
159  to_little(hash);
160 }

162 void sha256_next(const uint8_t data[], uint32_t length, uint32_t hash[8],
163  size_t total_length, uint8_t *tmpbuf) {
164  if (total_length == 0) {
165  sha256_stream(hash, data, length);
166  } else {
167  size_t sz = 0;
168  uint8_t *message = alloc_padded(length, total_length, &sz, tmpbuf);
169  memcpy(message, data, length);
170  sha256_stream(hash, data, sz);
171  if (message != tmpbuf)
172  free(message);
173  }

182 // calculate file SHA256
183 void sha256_file(const char *fname, uint32_t hash[8]) {
184  FILE *f = fopen(fname, "rb");
185  if (!f) {
186  fprintf(stderr, "Error opening file %s\n", fname);
187  exit(EXIT_FAILURE);
188  }
189  const size_t BUFSIZE = 0x1000000; // 16 MiB
190  char *buf = (char *)calloc(BUFSIZE, sizeof(char));
191  assert(buf);
192  init_hash(hash);
193  size_t length = 0;
194  while (1) {
195  size_t bytes = fread(buf, 1, BUFSIZE, f);
196  if (!bytes) {
197  perror("Error reading from file");
198  exit(EXIT_FAILURE);
199  }
200  int eof = 0;
201  if (bytes < BUFSIZE)
202  eof = 1;
203  else {
204  int c = getc(f);
205  ungetc(c, f);
206  if (c == EOF)
207  eof = 1;
208  }
209  if (eof) {
210  const uint64_t message_size = next_div_by(bytes + 1 + 8, 64);
211  // allocate and zero out
212  uint8_t *message = (uint8_t *)calloc(message_size, sizeof(char));
213  assert(buf);
214  memcpy(message, buf, bytes);
215  // pad with '1' and zeros (array already zeroed out)
216  message[bytes] = 0x80; // 100..
217  // pad with length in big endian format
218  const uint64_t data_bit_size = 8 * (length + bytes);
219  const uint64_t size = to_big_endian(data_bit_size);
220  memcpy(&message[message_size - 8], &size, sizeof(uint64_t));
221  sha256_stream(hash, (uint8_t *)message, message_size);
222  free(message);
223  break;
224  } else {
225 
226  sha256_stream(hash, (uint8_t *)buf, (uint32_t)BUFSIZE);
227  }
228  memset(buf, 0, BUFSIZE);
229  length += bytes;
230  }
231  to_little(hash);
232  free(buf);
233 }

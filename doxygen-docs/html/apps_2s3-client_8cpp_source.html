38  * \file s3-client.cpp
39  * \brief Send S3v4-signed requests to S3 andpoint and print raw responses.

41 /// [Generic S3 Client]
42 #include <algorithm>
43 #include <iostream>
44 #include <vector>
45 
46 #include "lyra/lyra.hpp"
47 #include "s3-client.h"
48 #include "url_utility.h"
49 
50 using namespace std;
51 using namespace sss;
52 
53 //------------------------------------------------------------------------------
54 int main(int argc, char const *argv[]) {
55  try {
56  S3ClientConfig args;
57  bool showHelp = false;
58  string data;
59  string params;
60  string headers;
61  auto cli =
62  lyra::help(showHelp).description(
63  "Send REST request with S3v4 signing") |
64  lyra::opt(args.accessKey,
65  "awsAccessKey")["-a"]["--access_key"]("AWS access key")
66  .optional() |
67  lyra::opt(args.secretKey,
68  "awsSecretKey")["-s"]["--secret_key"]("AWS secret key")
69  .optional() |
70  lyra::opt(args.endpoint,
71  "endpoint")["-e"]["--endpoint"]("End-point URL")
72  .required() |
73  lyra::opt(args.method, "method")["-m"]["--method"](
74  "HTTP method: get | put | post | delete | head")
75  .optional() |
76  lyra::opt(params, "params")["-p"]["--params"](
77  "URL request parameters. key1=value1;key2=...")
78  .optional() |
79  lyra::opt(args.bucket, "bucket")["-b"]["--bucket"]("Bucket name")
80  .optional() |
81  lyra::opt(args.key, "key")["-k"]["--key"]("Key name").optional() |
82  lyra::opt(data, "content")["-d"]["--data"](
83  "Value data if -F option not present, filename otherwise")
84  .optional() |
85  lyra::opt(headers, "headers")["-H"]["--headers"](
86  "URL request headers. header1:value1;header2:...")
87  .optional() |
88  lyra::opt(args.outfile,
89  "output file")["-o"]["--out-file"]("output file")
90  .optional() |
91  lyra::opt(args.signUrl, "signing url")["-S"]["--sign-url"](
92  "URL for signing; can be different from endpoint to support "
93  "tunnels")
94  .optional() |
95  lyra::opt(args.dataIsFileName)["-F"]["--data-is-filename"](
96  "Interpret 'data' field as file name and read data from file")
97  .optional();
98 
99  // Parse program arguments:
100  auto result = cli.parse({argc, argv});
101  if (!result) {
102  cerr << result.message() << endl;
103  cerr << cli << endl;
104  exit(1);
105  }
106  if (showHelp) {
107  cout << cli;
108  return 0;
109  }
110  if (!params.empty())
111  args.params = ParseParams(params);
112  if (!headers.empty())
113  args.headers = ParseHeaders(headers);
114  if (!data.empty()) {
115  std::copy(begin(data), end(data), back_inserter(args.data));
116  if (args.dataIsFileName) {
117  args.data.push_back('\0');
118  }
119  }
120  auto req = SendS3Request(args);
121  // Status code 20* = no error
122  cout << "Status: " << req.StatusCode() << endl << endl;
123  // Response body
124  vector<char> resp = req.GetResponseBody();
125  string t(begin(resp), end(resp));
126  cout << t << endl << endl;
127  // Response header
128  vector<char> h = req.GetResponseHeader();
129  string hs(begin(h), end(h));
130  cout << hs << endl;
131  return 0;
132  } catch (const exception &e) {
133  cerr << e.what() << endl;
134  return 1;
135  }
136 }

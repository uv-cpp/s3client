50 // Break url into {protocol, hostname, port}
51 URL ParseURL(const string &s) {
52  smatch m;
53  // minimal version of a URL parser: no enforcement of having the host
54  // section start and end with a word and requiring a '.' character
55  // separating the individual words.
56  regex e("\\s*(\\w+)://([0-9a-zA-Z-_\\.]+)(:(\\d+))?");
57  if (!regex_search(s, m, e, regex_constants::match_any)) {
58  return {};
59  }
60  if (m[4].str().empty()) {
61  return {-1, m[2], m[1]};
62  } else {
63  return {int(stoul(m[4])), m[2], m[1]};
64  }
65 }

68 // urlencode string
69 string UrlEncode(const std::string &s) {
70  std::ostringstream escaped;
71  escaped.fill('0');
72  escaped << std::hex;
73 
74  for (std::string::const_iterator i = s.begin(), n = s.end(); i != n; ++i) {
75  std::string::value_type c = (*i);
76  // Keep alphanumeric and other accepted characters intact
77  if (isalnum(c) || c == '-' || c == '_' || c == '.' || c == '~') {
78  escaped << c;
79  continue;
80  }
81 
82  // Any other characters are percent-encoded
83  escaped << std::uppercase;
84  escaped << '%' << setw(2) << int((unsigned char)c);
85  escaped << std::nouppercase;
86  }
87  return escaped.str();
88 }
89 //------------------------------------------------------------------------------
90 // Return urlencoded url request parameters from {key, value} dictionary
91 string UrlEncode(const Map &p) {
92  string url;
93  for (auto i : p) {
94  url += UrlEncode(i.first) + "=" + UrlEncode(i.second);
95  url += '&';
96  }
97  return string(url.begin(), --url.end());
98 }

103  std::transform(s.begin(), s.end(), s.begin(),
104  [](unsigned char c) { return std::toupper(c); });
105  return s;

110 string ToLower(string s) {
111  std::transform(s.begin(), s.end(), s.begin(),

119 // From "key1=value1;key2=value2;key3=;key4" to {key, value} dictionary
120 Parameters ParseParams(string s) {
121  if (s.empty())
122  return Parameters();
123  vector<string> slist;
124  Split(s, slist, ";");
125  Map params;
126  for (auto p : slist) {
127  vector<string> kv;
128  if (p.find_first_of("=") == string::npos)
129  p += "=";
130  Split(p, kv, "=", 1);
131  assert(kv.size() == 1 || kv.size() == 2);
132  const string key = kv[0];
133  const string value = kv.size() == 2 ? kv[1] : "";
134  params.insert({key, value});
135  }
136  return params;
137 }

139 //------------------------------------------------------------------------------
140 // From "key1:value1;key2:value2 {key, value} dictionary
141 // Per https://www.w3.org/Protocols/rfc2616/rfc2616.html
142 // HTTP headers are not case-sensitive and therefore it's better to translate
143 // all of them to lower-case to avoid problems when they are places is
144 // lexicographic order as part of the S3v4 signing process.
145 Headers ParseHeaders(const string &s) {
146  if (s.empty())
147  return Headers();
148  vector<string> slist;
149  Split(s, slist, ";");
150  Headers headers;
151  for (auto p : slist) {
152  vector<string> kv;
153  Split(p, kv, ":", 1);
154  assert(kv.size() == 1 || kv.size() == 2);
155  const string key = kv[0];
156  const string value = kv.size() == 2 ? kv[1] : "";
157  headers.insert({ToLower(key), value});
158  }
159  return headers;
160 }

162 //------------------------------------------------------------------------------
163 /// Adds \c x-amz-meta- prefix to map keys and checks that total size is less
164 /// than maximum metadata header size (currently 2kB).
165 Map ToMeta(const Map &metadata) {
166  const size_t MAX_META_SIZE = 2048;
167  Map meta;
168  const string prefix = "x-amz-meta-";
169  size_t size = 0;
170  for (auto kv : metadata) {
171  auto key = prefix + kv.first;
172  meta[key] = kv.second;
173  size += key.size() + kv.second.size() + 1; // +1 --> ":"
174  }
175  if (size > MAX_META_SIZE)
176  throw std::domain_error("Size of metadata bigger than " +
177  to_string(MAX_META_SIZE) + " bytes");
178  return meta;
179 }

181 //------------------------------------------------------------------------------
182 /// Map key to value using regular expression.
183 string GetValue(const map<string, string> &map, const string &rx,
184  bool caseSensitive) {
185  for (const auto &kv : map) {
186  regex re = caseSensitive ? regex(rx) : regex(rx, regex_constants::icase);
187  if (regex_match(kv.first, re)) {
188  return kv.second;
189  }
190  }
191  return string();
192 }

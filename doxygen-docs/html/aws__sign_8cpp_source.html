63 std::string SHA256(const string &s) {
64  uint32_t hash[8];
65  sha256::sha256((const uint8_t *)s.c_str(), (uint32_t)s.size(), hash);
66  char h[65];
67  hash_to_text(hash, h);
68  return h;
69 }

71 Bytes HMAC256(const Bytes &d, const Bytes &k) {
72  vector<uint8_t> hash(32);
73  hmac256(d.data(), d.size(), k.data(), k.size(), hash.data());
74  return hash;
75 }

77 /// Return time in the two formats required to sign AWS requests:
78 /// - full date-time
79 /// - date only
80 Time GetDates() {
81  time_t t;
82  time(&t);
83  struct tm *ts;
84  ts = gmtime(&t);
85 
86  const size_t BUFSIZE = 128;
87  vector<char> buf1(BUFSIZE, '\0');
88  strftime(buf1.data(), BUFSIZE, "%Y%m%dT%H%M%SZ", ts);
89  const string timeStamp(buf1.data());
90  vector<char> buf2(BUFSIZE, '\0');
91  strftime(buf2.data(), BUFSIZE, "%Y%m%d", ts);
92  const string dateStamp(buf2.data());
93  return {timeStamp, dateStamp};
94 }

101 /// Create AWS signature key as byte array
102 Bytes CreateSignatureKey(const string &key, const string &dateStamp,
103  const string &region, const string &service) {
104  const string k = "AWS4" + key;
105  const Bytes kb = Bytes(k.begin(), k.end());
106  const Bytes dsb = Bytes(dateStamp.begin(), dateStamp.end());
107  const Bytes dateKey = Hash(kb, dsb);
108  const Bytes regionKey = Hash(dateKey, Bytes(begin(region), end(region)));
109  const Bytes serviceKey = Hash(regionKey, Bytes(begin(service), end(service)));
110  const string aws = "aws4_request";
111  const Bytes signingKey = Hash(serviceKey, Bytes(begin(aws), end(aws)));
112  return signingKey;
113 }

126 //------------------------------------------------------------------------------
127 /// Presign url, 'expiration' time must be specified in seconds
128 /// @warnihg: x-amz- fields seem not to be required to sign, metatdata
129 /// x-amz-meta work without adding to signed headers
130 string SignedURL(const S3SignUrlConfig &cfg) {
131 
132  // const string &accessKey, const string &secretKey,
133  // int expiration, const string &endpoint, const string &method,
134  // const string &bucketName, const string &keyName,
135  // const Parameters &params, const string &region,
136  // const Time &dates) {
137  const URL url = ParseURL(cfg.endpoint);
138  const string host =
139  url.port <= 0 ? url.host : url.host + ":" + to_string(url.port);
140  Time t = cfg.dates.dateStamp.empty() ? GetDates() : cfg.dates;
141 
142  const string credentials =
143  cfg.access + "/" + t.dateStamp + "/" + cfg.region + "/s3/aws4_request";
144 
145  // headers: add x-amz headers to singned headers
146  Headers signHeaders = {{"host", host}};
147  for (auto kv : cfg.headers) {
148  if (kv.first.find("x-amz-")) {
149  signHeaders.insert(kv);
150  }
151  }
152  string signedHeadersStr;
153  for (auto kv : signHeaders) {
154  signedHeadersStr += kv.first + ';';
155  }
156  signedHeadersStr.pop_back(); // remove last ';'
157 
158  Headers headers = cfg.headers;
159  headers.insert({"host", host});
160 
161  ostringstream os;
162  for (auto kv : headers) {
163  os << kv.first << ':' << kv.second << '\n';
164  }
165 
166  const string canonicalHeaders = os.str();
167  ////
168 
169  Parameters parameters = {{"X-Amz-Algorithm", "AWS4-HMAC-SHA256"},
170  {"X-Amz-Credential", credentials},
171  {"X-Amz-Date", t.timeStamp},
172  {"X-Amz-Expires", to_string(cfg.expiration)},
173  {"X-Amz-SignedHeaders", signedHeadersStr}};
174 
175  if (!cfg.params.empty()) {
176  parameters.insert(begin(cfg.params), end(cfg.params));
177  }
178  const string canonicalQueryStringUrlEncoded = UrlEncode(parameters);
179 
180  string canonicalResource = "/";
181 
182  if (!cfg.bucket.empty()) {
183  canonicalResource += cfg.bucket;
184  if (!cfg.key.empty()) {
185  canonicalResource += "/" + cfg.key;
186  }
187  }
188 
189  const string payloadHash = "UNSIGNED-PAYLOAD";
190 
191  const string canonical_request = cfg.method + "\n" + canonicalResource +
192  "\n" + canonicalQueryStringUrlEncoded +
193  "\n" + canonicalHeaders + "\n" +
194  signedHeadersStr + "\n" + payloadHash;
195 
196  // text to sign
197  const string hashingAlgorithm = "AWS4-HMAC-SHA256";
198  const string credentialCtx =
199  t.dateStamp + "/" + cfg.region + "/" + "s3" + "/" + "aws4_request";
200  const string stringToSign = hashingAlgorithm + "\n" + t.timeStamp + "\n" +
201  credentialCtx + "\n" + SHA256(canonical_request);
202 
203  // generate the signature
204  const Bytes signatureKey =
205  CreateSignatureKey(cfg.secret, t.dateStamp, cfg.region, "s3");
206 
207  const string signature =
208  Hex(HMAC256(Bytes(begin(stringToSign), end(stringToSign)), signatureKey));
209 
210  string requestUrl = cfg.endpoint;
211  if (!cfg.bucket.empty()) {
212  requestUrl += "/" + cfg.bucket;
213  if (!cfg.key.empty()) {
214  requestUrl += "/" + cfg.key;
215  }
216  }
217  requestUrl +=
218  "?" + canonicalQueryStringUrlEncoded + "&X-Amz-Signature=" + signature;
219 
220  return requestUrl;
221 }

223 //------------------------------------------------------------------------------
224 // Sign HTTP headers: return dictionary with {key, value} pairs containing
225 // per-header information.
226 Signature ComputeSignature(const ComputeSignatureConfig &cfg) {
227 
228 #ifndef NDEBUG
229  // do not want to waste time converting to lowercase
230  for (auto kv : cfg.headers) {
231  if (kv.first != ToLower(kv.first)) {
232  throw invalid_argument("Header keys must be lowecase");
233  }
234  }
235 #endif
236  const string payloadHash =
237  cfg.payloadHash.empty() ? "UNSIGNED-PAYLOAD" : cfg.payloadHash;
238  const URL url = ParseURL(cfg.endpoint);
239  const string host =
240  url.port <= 0 ? url.host : url.host + ":" + to_string(url.port);
241  Time t = cfg.dates.dateStamp.empty() ? GetDates() : cfg.dates;
242  const string reqParameters =
243  cfg.parameters.empty() ? "" : UrlEncode(cfg.parameters);
244  string canonicalURI = "/";
245 
246  if (!cfg.bucket.empty()) {
247  canonicalURI += cfg.bucket;
248  if (!cfg.key.empty()) {
249  canonicalURI += "/" + cfg.key;
250  }
251  }
252 
253  const string canonicalQueryString = reqParameters;
254 
255  // Canonical and signed headers
256  const Map defaultHeaders = {{"host", host},
257  {"x-amz-content-sha256", payloadHash},
258  {"x-amz-date", t.timeStamp}};
259 
260  Map canonicalHeaders = defaultHeaders;
261  set<string> sortedSignedHeadersKeys;
262  for (auto kv : defaultHeaders) {
263  sortedSignedHeadersKeys.insert(kv.first);
264  }
265  // add x-amz headers
266  for (auto kv : cfg.headers) {
267  if (kv.first.find("x-amz-") == 0 || kv.first.find("content-length") == 0) {
268  canonicalHeaders.insert(kv);
269  sortedSignedHeadersKeys.insert(kv.first);
270  }
271  }
272 
273  // canonical headers string
274  ostringstream os;
275  for (auto kv : canonicalHeaders) {
276  os << kv.first << ":" << kv.second << "\n";
277  }
278 
279  string canonicalHeadersStr = os.str();
280 
281  os.str("");
282  for (auto k : sortedSignedHeadersKeys) {
283  os << k << ";";
284  }
285 
286  string signedHeadersStr = os.str();
287  signedHeadersStr = signedHeadersStr.substr(0, signedHeadersStr.size() - 1);
288 
289  // canonical request
290  const string canonicalRequest =
291  ToUpper(cfg.method) + '\n' + canonicalURI + '\n' + canonicalQueryString +
292  '\n' + canonicalHeadersStr + '\n' + signedHeadersStr + '\n' + payloadHash;
293 
294  const string algorithm = "AWS4-HMAC-SHA256";
295  const string credentialScope =
296  t.dateStamp + '/' + cfg.region + '/' + cfg.service + '/' + "aws4_request";
297 
298  const string stringToSign = algorithm + '\n' + t.timeStamp + '\n' +
299  credentialScope + '\n' + SHA256(canonicalRequest);
300  // generate the signature
301  const Bytes signatureKey =
302  CreateSignatureKey(cfg.secret, t.dateStamp, cfg.region, cfg.service);
303 
304  const auto s =
305  Hex(HMAC256(Bytes(begin(stringToSign), end(stringToSign)), signatureKey));
306  const string signature(begin(s), end(s));
307  return {signature, credentialScope, signedHeadersStr, defaultHeaders};
308 }

310 //------------------------------------------------------------------------------
311 /// Sign HTTP headers: return dictionary with {key, value} pairs containing
312 /// per-header information.
313 Headers SignHeaders(const ComputeSignatureConfig &cfg) {
314  const auto s = ComputeSignature(cfg);
315  const string algorithm = "AWS4-HMAC-SHA256";
316  // build authorisaton header
317  const string authorizationHeader =
318  algorithm + ' ' + string("Credential=") + cfg.access + '/' +
319  s.credentialScope + ", " + string("SignedHeaders=") + s.signedHeadersStr +
320  ", " + string("Signature=") + s.signature;
321  auto allHeaders = s.defaultHeaders;
322  allHeaders.insert({"Authorization", authorizationHeader});
323  allHeaders.insert(begin(cfg.headers), end(cfg.headers));
324  return allHeaders;
325 }

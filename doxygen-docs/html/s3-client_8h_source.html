84 /// \brief Parameters for SendS3Request calls.
85 /// \see SendS3Request
86 struct S3ClientConfig {
87  std::string accessKey; ///< access
88  std::string secretKey; ///< secret
89  std::string endpoint; ///< actual endpoint where requests are sent
90  std::string
91  signUrl; ///< url used to sign, allows requests to work across tunnels
92  std::string bucket; ///< bucket name
93  std::string key; ///< key name
94  Parameters params; ///< url parameters as {key, value} map
95  std::string method = "GET"; ///< HTTP method name
96  Headers headers; ///< HTTP header map {header name, header value}
97  std::string outfile; ///< if not empty stores returned response body into file
98  std::vector<char> data; ///< data buffer or file name
99  /// if dataIsFileName == true interpret content of \c data field as file
100  /// name else interpret as data source
101  bool dataIsFileName = false;
102 };

104 /// \brief Parameters for calls to upload and download functions.
105 struct S3DataTransferConfig {
106  std::string accessKey; ///< access
107  std::string secretKey; ///< secret
108  std::string proxyUrl; ///< @warning not implemented @todo implement
109  std::string bucket; ///< bucket name
110  std::string key; ///< key name
111  std::string file; ///< file name
112  char *data = nullptr; ///< input for upload and output for download
113  size_t offset = 0; ///< input/ouput offset in buffer
114  size_t size = 0; ///< data size
115  std::vector<std::string>
116  endpoints; ///< list of endpoints for client-side load balancing
117  int maxRetries = 1; ///< maximum number of retries
118  int jobs = 1; ///< number of parallel upload/download tasks
119  size_t partsPerJob = 1; ///< number of parts per job
120  std::string payloadHash; ///< payload hash if empty the literal \c
121  ///< "UNSIGNED-PAYLOAD" is used instead of the SHA256
122  ///< hash code
123 };

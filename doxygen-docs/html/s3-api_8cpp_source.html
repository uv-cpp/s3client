42 /// [WebClient::Config]
43 WebClient &S3Api::Config(const SendParams &p) {
44  // if credentials empty send regular unsigned request
45  auto sh = Access().empty() ? Headers()
46  : SignHeaders({.access = Access(),
47  .secret = Secret(),
48  .endpoint = Endpoint(),
49  .method = p.method,
50  .bucket = p.bucket,
51  .key = p.key,
52  .payloadHash = p.payloadHash,
53  .parameters = p.params,
54  .headers = p.headers,
55  .region = p.region});
56  std::string path;
57  if (!p.bucket.empty()) {
58  path += "/" + p.bucket;
59  if (!p.key.empty()) {
60  path += "/" + p.key;
61  }
62  }
63  Clear();
64  webClient_.SetEndpoint(Endpoint());
65  webClient_.SetPath(path);
66  webClient_.SetMethod(p.method);
67  webClient_.SetReqParameters(p.params);
68  webClient_.SetHeaders(sh);
69  return webClient_;
70 }

73 ssize_t S3Api::GetObjectSize(const string &bucket, const string &key) {
74  const bool caseInsensitive = false;
75  try {
76  const string v =
77  GetValue(HeadObject(bucket, key), "length", caseInsensitive);
78  // if object exists header should always be present, but just in case...
79  return v.empty() ? -1 : stoll(v);
80  } catch (...) {
81  return -1;
82  }
83 }

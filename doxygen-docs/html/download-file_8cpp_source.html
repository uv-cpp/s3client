56 //-----------------------------------------------------------------------------
57 void DownloadPart(S3Api &s3, const string &file, const string &bucket,
58  const string &key, size_t offset, size_t partSize,
59  int maxRetries) {
60  try {
61  s3.GetFileObject(file, bucket, key, offset, offset, offset + partSize - 1);
62  } catch (const exception &e) {
63  if (retriesG++ > maxRetries)
64  throw e;
65  else
66  DownloadPart(s3, file, bucket, key, offset, partSize, maxRetries);
67  }
68 }

70 //-----------------------------------------------------------------------------
71 void DownloadPart(S3Api &s3, char *data, const string &bucket,
72  const string &key, size_t offset, size_t partSize,
73  int maxRetries) {
74  try {
75  s3.GetObject(bucket, key, data, offset, offset, offset + partSize - 1);
76  } catch (const exception &e) {
77  if (retriesG++ > maxRetries)
78  throw e;
79  else
80  DownloadPart(s3, data, bucket, key, offset, partSize, maxRetries);
81  }
82 }
83 //-----------------------------------------------------------------------------
84 void DownloadParts(const S3DataTransferConfig &cfg, size_t chunkSize,
85  int firstPart, int lastPart, size_t objectSize, int jobId) {
86  size_t offset = jobId * chunkSize;
87  chunkSize = min(chunkSize, objectSize - offset);
88  const int numParts = lastPart - firstPart;
89  const size_t partSize = (chunkSize + numParts - 1) / numParts;
90  const auto endpoint = cfg.endpoints[RandomIndex(0, cfg.endpoints.size() - 1)];
91  S3Api s3(cfg.accessKey, cfg.secretKey, endpoint);
92  for (int i = 0; i != numParts; ++i) {
93  const size_t size = min(partSize, chunkSize - i * partSize);
94  DownloadPart(s3, cfg.data ? cfg.data : cfg.file, cfg.bucket, cfg.key,
95  offset, size, cfg.maxRetries);
96  offset += size;
97  }
98 }

100 //-----------------------------------------------------------------------------
101 void DownloadFile(const S3DataTransferConfig &cfg, bool sync) {
102  retriesG = 0;
103  if (cfg.endpoints.empty()) {
104  throw std::logic_error("No endpoint specified");
105  }
106  S3Api s3(cfg.accessKey, cfg.secretKey, cfg.endpoints[0]);
107  const size_t fileSize = s3.GetObjectSize(cfg.bucket, cfg.key);
108  // create output file
109  std::ofstream ofs(cfg.file, std::ios::binary | std::ios::out);
110  ofs.seekp(fileSize - 1);
111  ofs.write("", 1);
112  ofs.close();
113  // initiate request
114  const size_t perJobSize = (fileSize + cfg.jobs - 1) / cfg.jobs;
115  // send parts in parallel and store ETags
116  vector<future<void>> dloads(cfg.jobs);
117  for (int i = 0; i != cfg.jobs; ++i) {
118  dloads[i] = async(sync ? launch::deferred : launch::async, DownloadParts,
119  cfg, perJobSize, i * cfg.partsPerJob,
120  i * cfg.partsPerJob + cfg.partsPerJob, fileSize, i);
121  }
122  for (auto &i : dloads) {
123  i.wait();
124  }
125 }

127 //-----------------------------------------------------------------------------
128 void DownloadData(const S3DataTransferConfig &cfg, bool sync) {
129  retriesG = 0;
130  if (!cfg.data) {
131  throw logic_error("NULL data buffer");
132  }
133  if (cfg.endpoints.empty()) {
134  throw std::logic_error("No endpoint specified");
135  }
136  S3Api s3(cfg.accessKey, cfg.secretKey, cfg.endpoints[0]);
137  // initiate request
138  const size_t perJobSize = (cfg.size + cfg.jobs - 1) / cfg.jobs;
139  // send parts in parallel and store ETags
140  vector<future<void>> dloads(cfg.jobs);
141  for (int i = 0; i != cfg.jobs; ++i) {
142  dloads[i] = async(sync ? launch::deferred : launch::async, DownloadParts,
143  cfg, perJobSize, i * cfg.partsPerJob,
144  i * cfg.partsPerJob + cfg.partsPerJob, cfg.size, i);
145  }
146  for (auto &i : dloads) {
147  i.wait();
148  }
149 }

151 //-----------------------------------------------------------------------------
152 void Download(const S3DataTransferConfig &cfg, bool sync) {
153  if (!cfg.data && cfg.file.empty()) {
154  throw logic_error("File name and data buffer pointer are both NULL");
155  }
156  if (cfg.data) {
157  DownloadData(cfg, sync);
158  } else {
159  DownloadFile(cfg, sync);
160  }
